db.createCollection("user", {
   validator: {
      $jsonSchema: {
         bsonType: "object",
         required: [ "name", "age"],
         properties: {
            name: {
               bsonType: "string",
               description: "must be a string and is required"
            },
            age: {
               bsonType: "int",
               minimum: 0,
               maximum: 150,
               description: "must be an integer in [ 0, 150 ] and is required"
            },
         }
      }
   }
})
